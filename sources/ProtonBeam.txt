# PROTON BEAM SOURCE
# Momentum specified via parameter override in run files
# Default: 1 GeV/c for testing
#------------------------------------------------------------------------------
s:So/ProtonBeam/Type                     = "Beam"
s:So/ProtonBeam/Component                = "World"
s:So/ProtonBeam/BeamParticle             = "proton"

# Momentum (GeV/c) - will be overridden per run

# Energy spread (relative, ÏƒE/E ~ 1% for T09)

# Angular distribution (Gaussian with ~1 mrad divergence)
s:So/ProtonBeam/BeamAngularDistribution  = "Gaussian"

d:So/ProtonBeam/BeamAngularSpreadY       = 1 mrad
d:So/ProtonBeam/BeamAngularCutoffX       = 5 mrad
d:So/ProtonBeam/BeamAngularCutoffY       = 5 mrad

# Number of histories (will be overridden per run)
i:So/ProtonBeam/NumberOfHistoriesInRun   = 10000
